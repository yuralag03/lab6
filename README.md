	if ((x == 1 && y == 1) && (x == 1 && y == 2) && (x == 1 && y == 3) && (x == 1 && y == 5) && (x == 1 && y == 6) && (x == 1 && y == 7) &&

		(x == 6 && y == 1) && (x == 6 && y == 2) && (x == 6 && y == 5) && (x == 6 && y == 7) &&
